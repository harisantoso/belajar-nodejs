menggunakan package


cara instal package npm
co: "npm install namaPackage@versiPackage" 
> npm install chalk@4.1.0 

cara uninstal package npm
co: "npm uninstal namaPackage@versiPackage"
> npm uninstall chalk@4.1.0

nodemon version 2.0.7 // secara global

ctrl + c untuk stop nodemon 

cara instal nodemon secara global
> npm uninstall -g nodemon@2.0.7

cara instal nodemon secara local
> npm uninstall --save-dev nodemon@2.0.7

cara uninstall nodemon
> npm uninstall -g nodemon

npm list -g // secara global
